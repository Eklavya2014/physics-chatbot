DATASTORE (Save Player Data):
local DataStoreService = game:GetService("DataStoreService")
local playerData = DataStoreService:GetDataStore("PlayerData")

local function saveData(player)
    local data = {
        coins = player.leaderstats.Coins.Value,
        level = player.leaderstats.Level.Value
    }
    local success, err = pcall(function()
        playerData:SetAsync("Player_"..player.UserId, data)
    end)
    if not success then warn("Save failed:", err) end
end

local function loadData(player)
    local success, data = pcall(function()
        return playerData:GetAsync("Player_"..player.UserId)
    end)
    if success and data then
        player.leaderstats.Coins.Value = data.coins or 0
        player.leaderstats.Level.Value = data.level or 1
    end
end

SQL - COMPLETE DATABASE GUIDE

SQL BASICS:
SQL (Structured Query Language) for relational databases.
Databases: PostgreSQL, MySQL, SQLite, SQL Server, Oracle.
Case insensitive for keywords. Statements end with ;

DATA TYPES:
INTEGER/INT, BIGINT, SMALLINT
DECIMAL(p,s), NUMERIC(p,s), FLOAT, DOUBLE PRECISION, REAL
VARCHAR(n), CHAR(n), TEXT
DATE, TIME, TIMESTAMP, TIMESTAMPTZ
BOOLEAN, UUID, JSON, JSONB (PostgreSQL)
BLOB, BYTEA (binary)

DDL (Data Definition Language):
CREATE TABLE users (
    id          SERIAL PRIMARY KEY,
    email       VARCHAR(200) UNIQUE NOT NULL,
    username    VARCHAR(50) NOT NULL,
    age         INTEGER CHECK(age >= 0 AND age <= 150),
    salary      DECIMAL(10,2) DEFAULT 0.00,
    is_active   BOOLEAN DEFAULT true,
    created_at  TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    dept_id     INTEGER REFERENCES departments(id) ON DELETE SET NULL
);

CREATE INDEX idx_users_email ON users(email);
CREATE INDEX idx_users_dept ON users(dept_id);
CREATE UNIQUE INDEX idx_users_username ON users(username);

ALTER TABLE users ADD COLUMN phone VARCHAR(20);
ALTER TABLE users DROP COLUMN phone;
ALTER TABLE users ALTER COLUMN salary SET DEFAULT 1000;
ALTER TABLE users RENAME COLUMN username TO user_name;
ALTER TABLE users ADD CONSTRAINT fk_dept FOREIGN KEY(dept_id) REFERENCES departments(id);

DROP TABLE IF EXISTS users;
TRUNCATE TABLE users;  -- delete all rows, reset auto-increment

DML (Data Manipulation Language):
-- INSERT
INSERT INTO users (email, username, age) VALUES ('alice@example.com', 'alice', 25);
INSERT INTO users VALUES (DEFAULT, 'bob@example.com', 'bob', 30, DEFAULT, DEFAULT, DEFAULT, 1);
INSERT INTO users (email, username) SELECT email, name FROM old_users WHERE active=true;

-- SELECT
SELECT * FROM users;
SELECT id, email, age FROM users;
SELECT DISTINCT dept_id FROM users;
SELECT u.email, d.name as dept_name FROM users u JOIN departments d ON u.dept_id = d.id;
SELECT *, ROW_NUMBER() OVER(PARTITION BY dept_id ORDER BY salary DESC) as rank FROM users;

-- WHERE conditions
SELECT * FROM users WHERE age > 25 AND is_active = true;
SELECT * FROM users WHERE age BETWEEN 20 AND 30;
SELECT * FROM users WHERE email LIKE '%@gmail.com';
SELECT * FROM users WHERE email ILIKE '%gmail%';  -- case insensitive (PostgreSQL)
SELECT * FROM users WHERE dept_id IN (1, 2, 3);
SELECT * FROM users WHERE dept_id IS NULL;
SELECT * FROM users WHERE dept_id IS NOT NULL;

-- UPDATE
UPDATE users SET salary = salary * 1.1, is_active = true WHERE dept_id = 1;
UPDATE users SET age = age + 1 WHERE email = 'alice@example.com' RETURNING *;

-- DELETE
DELETE FROM users WHERE is_active = false;
DELETE FROM users WHERE created_at < NOW() - INTERVAL '1 year';

JOINS:
-- INNER JOIN: only matching rows
SELECT u.name, d.dept_name FROM users u
INNER JOIN departments d ON u.dept_id = d.id;

-- LEFT JOIN: all from left, matching from right (NULL if no match)
SELECT u.name, d.dept_name FROM users u
LEFT JOIN departments d ON u.dept_id = d.id;

-- RIGHT JOIN: all from right table
-- FULL OUTER JOIN: all from both tables

-- SELF JOIN
SELECT a.name as employee, b.name as manager FROM users a
LEFT JOIN users b ON a.manager_id = b.id;

-- CROSS JOIN: all combinations
SELECT p.name, c.name FROM products p CROSS JOIN categories c;

AGGREGATION:
SELECT dept_id, COUNT(*) as count, AVG(salary) as avg_salary,
       MAX(salary) as max_sal, MIN(salary) as min_sal, SUM(salary) as total
FROM users
GROUP BY dept_id
HAVING COUNT(*) > 5
ORDER BY avg_salary DESC
LIMIT 10 OFFSET 20;

-- WINDOW FUNCTIONS
SELECT name, salary,
    RANK() OVER(ORDER BY salary DESC) as rank,
    DENSE_RANK() OVER(ORDER BY salary DESC) as dense_rank,
    ROW_NUMBER() OVER(PARTITION BY dept_id ORDER BY salary DESC) as dept_rank,
    LAG(salary) OVER(ORDER BY created_at) as prev_salary,
    LEAD(salary) OVER(ORDER BY created_at) as next_salary,
    AVG(salary) OVER(PARTITION BY dept_id) as dept_avg,
    SUM(salary) OVER(ORDER BY created_at ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW) as running_total
FROM users;

-- SUBQUERIES
SELECT name FROM users WHERE salary > (SELECT AVG(salary) FROM users);
SELECT * FROM users WHERE dept_id IN (SELECT id FROM departments WHERE budget > 100000);

-- CTE (Common Table Expression)
WITH high_earners AS (
    SELECT * FROM users WHERE salary > 80000
),
dept_stats AS (
    SELECT dept_id, AVG(salary) avg_sal FROM users GROUP BY dept_id
)
SELECT h.name, h.salary, d.avg_sal FROM high_earners h
JOIN dept_stats d ON h.dept_id = d.dept_id
ORDER BY h.salary DESC;

-- RECURSIVE CTE
WITH RECURSIVE org_chart AS (
    SELECT id, name, manager_id, 0 as level FROM employees WHERE manager_id IS NULL
    UNION ALL
    SELECT e.id, e.name, e.manager_id, oc.level+1 FROM employees e
    JOIN org_chart oc ON e.manager_id = oc.id
)
SELECT * FROM org_chart ORDER BY level;

fn main() {
    let x: i32 = 5;           // immutable
    let mut y = 10;           // mutable
    let z = x + y;
    println!("{} {} {}", x, y, z);
    
    // Ownership
    let s1 = String::from("hello");
    let s2 = s1;              // s1 moved, no longer valid
    let s3 = s2.clone();      // deep copy
    
    // References (borrowing)
    let len = calculate_length(&s2);  // borrow
    fn calculate_length(s: &String) -> usize { s.len() }
    
    // Slices
    let s = String::from("hello world");
    let hello = &s[0..5];
    
    // Structs
    #[derive(Debug, Clone)]
    struct Point { x: f64, y: f64 }
    impl Point {
        fn new(x: f64, y: f64) -> Self { Point {x, y} }
        fn distance(&self) -> f64 { (self.x*self.x+self.y*self.y).sqrt() }
    }
    
    // Enums
    enum Shape { Circle(f64), Rectangle(f64,f64), Triangle(f64,f64,f64) }
    impl Shape {
        fn area(&self) -> f64 {
            match self {
                Shape::Circle(r) => std::f64::consts::PI * r * r,
                Shape::Rectangle(w,h) => w*h,
                Shape::Triangle(a,b,c) => { let s=(a+b+c)/2.0; (s*(s-a)*(s-b)*(s-c)).sqrt() }
            }
        }
    }
    
    // Error handling with Result and Option
    let result: Result<i32,String> = Ok(42);
    let n = result.unwrap_or(0);
    let opt: Option<i32> = Some(5);
    if let Some(v) = opt { println!("{}", v); }
    
    // Traits
    trait Greet { fn hello(&self) -> String; }
    impl Greet for Point { fn hello(&self) -> String { format!("({},{})",self.x,self.y) } }
    
    // Closures
    let add = |a: i32, b: i32| a + b;
    let nums = vec![1,2,3,4,5];
    let squares: Vec<i32> = nums.iter().map(|x| x*x).collect();
    let evens: Vec<&i32> = nums.iter().filter(|&&x| x%2==0).collect();
    let sum: i32 = nums.iter().sum();
}

fun main() {

    sealed class Shape { class Circle(val r:Double):Shape(); class Rect(val w:Double,val h:Double):Shape() }
    fun area(s:Shape):Double = when(s) { is Shape.Circle -> Math.PI*s.r*s.r; is Shape.Rect -> s.w*s.h }
