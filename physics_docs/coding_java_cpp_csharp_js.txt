class Animal {
private:
    string name;
    int age;
    static int count;

public:
    // Constructors
    Animal(string name, int age) : name(name), age(age) { count++; }
    Animal(const Animal& other) : name(other.name), age(other.age) {}  // copy
    Animal(Animal&& other) noexcept : name(move(other.name)), age(other.age) {}  // move
    
    // Destructor
    virtual ~Animal() { count--; }
    
    // Getters
    string getName() const { return name; }
    
    // Virtual function (polymorphism)

        return *this;
    }
    friend ostream& operator<<(ostream& os, const Animal& a) {
        return os << "Animal(" << a.name << ")";
    }
    
    static int getCount() { return count; }
};

class Animal {
    #name;  // private field (ES2022)
    static count = 0;
    
    constructor(name, age) {
        this.#name = name;
        this.age = age;
        Animal.count++;
    }
    
    get name() { return this.#name; }
    set name(val) { if(val) this.#name = val; }
    
    speak() { return `${this.#name} makes a sound`; }
    static getCount() { return Animal.count; }
    
    toString() { return `Animal(${this.#name})`; }
}

class Dog extends Animal {
    constructor(name, age, breed) {
        super(name, age);
        this.breed = breed;
    }
    speak() { return `${this.name} says Woof!`; }
}

async function fetchUser(id) {
    try {
        const response = await fetch(`/api/users/${id}`);
        if (!response.ok) throw new Error(`HTTP ${response.status}`);
        const data = await response.json();
        return data;
    } catch(err) {
        console.error("Fetch failed:", err);
        throw err;
    }
}
